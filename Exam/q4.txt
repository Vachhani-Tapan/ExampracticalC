// q4 Vowel or Consonant

#include<stdio.h>
int  main()
{
    char c;
    printf("Enter any character here : ");
    scanf("%c",&c);

    if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u' || c=='A' || c=='E' 
    || c=='I' || c=='O' || c=='U')
    {
        printf("The Entered character is VOWEL");
    }
    else
    {
        printf("The Entered character is Consonant");
    }
    return 0;
}
